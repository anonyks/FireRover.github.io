#include

// Define motor pins
#dêfine LeftMotorForward 2
#dêfine LeftMotorBackward 3
#dêfine RightMotorForward 7
#dêfine RightMotorBackward 4
#dêfine SpeedA 5
#dêfine SpeedB 6

// Define flame sensor pins
#dêfine FLAME_SENSOR1 12
#dêfine FLAME_SENSOR2 9
#dêfine FLAME_SENSOR3 8

// Define water relay pin
const int WATER_RELAY = 13;

// Define ultrasonic sensor pins
#dêfine TRIGGER_PIN 5
#dêfine ECHO_PIN 6

// Define servo offsets
#dêfine OFFSET_LEFT -45
#dêfine OFFSET_RIGHT 45

// Declare servo objects
Servo servoMotor;
Servo waterServo;

// Declare variable for pulse duration
long pulseDuration;

// Declare variables for distances
int distance, centerDistance, rightDistance, leftDistance;

void setup() {
// Initialize motor pins
pinMode(RightMotorForward, OUTPUT);
pinMode(LeftMotorForward, OUTPUT);
pinMode(LeftMotorBackward, OUTPUT);
pinMode(RightMotorBackward, OUTPUT);
pinMode(SpeedA, OUTPUT);
pinMode(SpeedB, OUTPUT);

// Initialize ultrasonic sensor pins
pinMode(TRIGGER_PIN, OUTPUT);
pinMode(ECHO_PIN, INPUT);

// Initialize flame sensor pins
pinMode(FLAME_SENSOR1, INPUT);
pinMode(FLAME_SENSOR2, INPUT);
pinMode(FLAME_SENSOR3, INPUT);

// Initialize water relay pin
pinMode(WATER_RELAY, OUTPUT);
digitalWrite(WATER_RELAY, HIGH);

// Attach servo motors
servoMotor.attach(10);
waterServo.attach(11);

// Set initial servo positions
servoMotor.write(120);
waterServo.write(120);
}

void loop() {
// Check for fire
while (digitalRead(FLAME_SENSOR1) == LOW || digitalRead(FLAME_SENSOR2) == LOW || digitalRead(FLAME_SENSOR3) == LOW) {
off(); // Stop the robot
extinguishFire(); // Extinguish the fire
}

// Measure center distance
centerDistance = ultrasonic();

// Check for obstacles
if (centerDistance > 30) {
forward(); // Move forward if no obstacle
} else {
if (centerDistance <= 10 && centerDistance > 2) {
reverse(); // Reverse if obstacle is too close
off(); // Stop the robot
servoMotor.write(80); // Turn head to right
delay(300);
rightDistance = ultrasonic(); // Measure right distance
servoMotor.write(160); // Turn head to left
delay(300);
leftDistance = ultrasonic(); // Measure left distance
servoMotor.write(120); // Return head to center
delay(300);
comparison(rightDistance, leftDistance); // Decide which way to turn
off(); // Stop the robot
delay(150);
}
}

// Reset distances
rightDistance = 0;
leftDistance = 0;
centerDistance = 0;
}

// Function to extinguish fire
void extinguishFire() {
int initialPos = 120;
int offset = 0;

// Check which sensor detected the fire
if (digitalRead(FLAME_SENSOR1) == LOW) {
offset = OFFSET_LEFT;
} else if (digitalRead(FLAME_SENSOR3) == LOW) {
offset = OFFSET_RIGHT;
}

// Aim water servo at fire
waterServo.write(initialPos + offset);
delay(333);

// Start spraying water
digitalWrite(WATER_RELAY, LOW);

// Move water servo to cover more area
for (int i = 0; i <= 15; i++) {
waterServo.write(initialPos + offset + i);
delay(15);
}
for (int i = 15; i >= -15; i--) {
waterServo.write(initialPos + offset + i);
delay(15);
}
for (int i = -15; i <= 0; i++) {
waterServo.write(initialPos + offset + i);
delay(15);
}

// Stop spraying water
delay(1111);
digitalWrite(WATER_RELAY, HIGH);

// Return water servo to initial position
waterServo.write(initialPos);
}

// Function to measure distance using ultrasonic sensor
int ultrasonic(void) {
int distance = 0;

// Send pulse
digitalWrite(TRIGGER_PIN, LOW);
delayMicroseconds(2);
digitalWrite(TRIGGER_PIN, HIGH);
delayMicroseconds(10);
digitalWrite(TRIGGER_PIN, LOW);

// Measure pulse duration
pulseDuration = pulseIn(ECHO_PIN, HIGH);

// Calculate distance
distance = pulseDuration * 0.034 / 2;

return distance;
}

// Function to decide which way to turn
void comparison(int r, int l) {
if (r > 25 || l > 25) {
if (r > l || r == l) {
TurnRight90(); // Turn right if right is clear or both are clear
} else if (l > r) {
TurnLeft90(); // Turn left if left is clear
}
} else if (r < 25 && l < 25) {
TurnLeft180(); // Turn around if both sides are blocked
}
}

// Function to move forward
void forward() {
digitalWrite(SpeedA, 169);
digitalWrite(SpeedB, 169);
digitalWrite(RightMotorForward, HIGH);
digitalWrite(RightMotorBackward, LOW);
digitalWrite(LeftMotorForward, HIGH);
digitalWrite(LeftMotorBackward, LOW);
delay(500);
}

// Function to stop the robot
void off() {
digitalWrite(RightMotorForward, LOW);
digitalWrite(RightMotorBackward, LOW);
digitalWrite(LeftMotorForward, LOW);
digitalWrite(LeftMotorBackward, LOW);
}

// Function to turn left 90 degrees
void TurnLeft90() {
digitalWrite(SpeedA, 160);
digitalWrite(SpeedB, 150);
digitalWrite(RightMotorForward, HIGH);
digitalWrite(RightMotorBackward, LOW);
digitalWrite(LeftMotorForward, LOW);
digitalWrite(LeftMotorBackward, HIGH);
delay(650);
}

// Function to turn right 90 degrees
void TurnRight90() {
digitalWrite(SpeedA, 133);
digitalWrite(SpeedB, 144);
digitalWrite(RightMotorForward, LOW);
digitalWrite(RightMotorBackward, HIGH);
digitalWrite(LeftMotorForward, HIGH);
digitalWrite(LeftMotorBackward, LOW);
delay(750);
}

// Function to turn left 180 degrees
void TurnLeft180() {
digitalWrite(SpeedA, 169);
digitalWrite(SpeedB, 169);
digitalWrite(RightMotorForward, HIGH);
digitalWrite(RightMotorBackward, LOW);
digitalWrite(LeftMotorForward, LOW);
digitalWrite(LeftMotorBackward, HIGH);
delay(1500); // Delay to allow the robot to complete the turn
}

// Function to move the robot in reverse
void reverse() {
// Check for fire while reversing
while (digitalRead(FLAME_SENSOR1) == LOW || digitalRead(FLAME_SENSOR2) == LOW || digitalRead(FLAME_SENSOR3) == LOW) {
off(); // Stop the robot
extinguishFire();
}

digitalWrite(SpeedA, 140);
digitalWrite(SpeedB, 140);
digitalWrite(RightMotorForward, LOW);
digitalWrite(RightMotorBackward, HIGH);
digitalWrite(LeftMotorForward, LOW);
digitalWrite(LeftMotorBackward, HIGH);
delay(400); // Delay to allow the robot to move a certain distance
}
